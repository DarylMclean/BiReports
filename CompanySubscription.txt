	<queryString>
		<![CDATA[select
    d.subNo, d.current, d.expired, d.future, d.OverallCompliance, d.*
from
(
	select
	    *,
	    @subNo := IF(@lastCompanyId = d.companyId, @subNo + 1, 1) As subNo,
	    @lastCompanyId := d.companyId
	from
	(
		select 
            case
                when d.current = true and d.OverallCompliance = 'C' then 1
                when d.future = true and d.OverallCompliance = 'C' then 2
                when d.current = true and d.OverallCompliance != 'C' then 3
                when d.future = true and d.OverallCompliance != 'C' then 4
                else 99
            end as score,
		    d.*
		from
		(
			select distinct
			    d.*,

                case
                    when now() > d.toDate then 'NC'
                    when pendingCompanyComplianceCertsId = 0 and profileStatus = 'C' then 'C'
                    when   (insuranceCompliant = 'N/A' or insuranceCompliant = 'C' or insuranceCompliant = '1' or insuranceCompliant = '2' or insuranceCompliant = '3')
                        and (licenceCompliant = 'N/A' or licenceCompliant = 'C' or licenceCompliant = '1' or licenceCompliant = '2' or licenceCompliant = '3')
                        and (SmsCompliant = 'N/A' or SmsCompliant = 'C' or SmsCompliant = '1' or SmsCompliant = '2' or SmsCompliant = '3')
                        and profileStatus = 'C' then 'C'
                    else 'NC'
                end OverallCompliance,
                if(now() between d.fromDate and d.toDate, 1, 0) as current,
                if(now() > d.toDate, 1, 0) as expired,
                if(now() < d.fromDate, 1, 0) as future
			from
			(
				select
				    ComplianceData.*,
                    case
                        when pendingCompanyComplianceCertsId is null and smsReviewInProgress IN ('S','SU','R') then smsReviewInProgress

                        when (certCompliant='1' or SmsReviewCompliant = '1') and (certCompliant<>'X' or SmsReviewCompliant<>'X') then '1'
                        when (certCompliant='2' or SmsReviewCompliant = '2') and (certCompliant<>'X' or SmsReviewCompliant<>'X') then '2'
                        when (certCompliant='3' or SmsReviewCompliant = '3') and (certCompliant<>'X' or SmsReviewCompliant<>'X') then '3'
                        when certCompliant='' and SmsReviewCompliant ='' then ''
                        when certCompliant='' and SmsReviewCompliant ='X' then 'X'
                        when certCompliant='X' and SmsReviewCompliant ='' then 'X'
                        when certCompliant='C' and SmsReviewCompliant ='X' then 'X'
                        when certCompliant='X' and SmsReviewCompliant ='C' then 'X'
                        when certCompliant='C' and SmsReviewCompliant ='C' then 'C'
                        when certCompliant='' and SmsReviewCompliant ='C' then 'C'
                        when certCompliant='C' and SmsReviewCompliant ='' then 'C'
                        else 'X'
                    end registeredCompliant,
                    case
                        when requiresSmsReview = 'n' then 'N/A'
                        when smsReviewInProgress IN ('S','SU','R') then smsReviewInProgress
                        when (SmsReviewCompliant = '1' or SMS4801 = '1') and SmsReviewCompliant<>'X' and SMS4801 <> 'X' then '1' 
                        when (SmsReviewCompliant = '2' or SMS4801 = '2') and SmsReviewCompliant<>'X' and SMS4801 <> 'X' then '2' 
                        when (SmsReviewCompliant = '3' or SMS4801 = '3') and SmsReviewCompliant<>'X' and SMS4801 <> 'X' then '3' 
                        when SmsReviewCompliant = '' and SMS4801 = '' then ''
                        when SmsReviewCompliant = '' and SMS4801 = 'X' then 'X'
                        when SmsReviewCompliant = 'X' and SMS4801 ='' then 'X'
                        when SmsReviewCompliant = 'C' and SMS4801 = 'X' then 'X'
                        when SmsReviewCompliant = 'X' and SMS4801 = 'C' then 'X'
                        when SmsReviewCompliant = 'C' and (SMS4801 = 'C' or SMS4801 = 'N/A') then 'C'
                        when SmsReviewCompliant = '' and (SMS4801 = 'C' or SMS4801 = 'N/A') then 'C'
                        when SmsReviewCompliant = 'C' and (SMS4801 = '' or SMS4801 = 'N/A') then 'C'
                        else 'X'
                    end SmsCompliant
				from
				(
					select
					    P.name portalName,
                        CCS.companyId, CCS.id, CCS.fromDate, CCS.toDate,
                        CO.name companyName,
                        CO.abn,
                        CCC.name categoryName,
                        case
                            when now() > CCS.toDate then 'X'
                            when PCCS.status = 'started' then 'S'
                            when PCCS.status = 'submitted' then 'SU'
                            when PCCS.status = 'returned' then 'R'
                            when PCCS.status = 'approved' then 'C'
                            else ''
                        end as profileStatus,
                        CCS.pendingCompanyComplianceCertsId,
                        case
                            when CCS.pendingCompanyComplianceCertsId is null or coalesce(Y.requiredCertCount, 0) = 0 then 'N/A'
                       
                            when X.minCertExpiryDays<30 and X.actualCertCount=Y.requiredCertCount and PCCC.status='approved' then '1'
                            when X.minCertExpiryDays<60 and X.actualCertCount=Y.requiredCertCount and PCCC.status='approved' then '2'
                            when X.minCertExpiryDays<90 and X.actualCertCount=Y.requiredCertCount and PCCC.status='approved' then '3'
                            when CCS.pendingCompanyComplianceCertsId >0 and coalesce(X.actualCertCount, 0) >= coalesce(Y.requiredCertCount, 0) and PCCC.status='approved' then 'C'
                            when PCCC.status='started' then 'S'
                            when PCCC.status = 'submitted' then 'SU'
                            when PCCC.status = 'returned' then 'R'
                            else 'X'
                        end certCompliant,
                     
                        case
                            when PCCC.status='started' then 'S'
                            when PCCC.status = 'submitted' then 'SU'
                            when PCCC.status = 'returned' then 'R'
                            when CCS.pendingCompanyComplianceCertsId is null or (coalesce(Y.requiredLicenceCount, 0) = 0 and PCCC.status = 'approved') then 'N/A'
                      
                            when X.minLicenceExpiryDays<30 and X.actualLicenceCount=Y.requiredLicenceCount and PCCC.status='approved' then '1'
                            when X.minLicenceExpiryDays<60 and X.actualLicenceCount=Y.requiredLicenceCount and PCCC.status='approved' then '2'
                            when X.minLicenceExpiryDays<90 and X.actualLicenceCount=Y.requiredLicenceCount and PCCC.status='approved' then '3'
                            when CCS.pendingCompanyComplianceCertsId >0 and coalesce(X.actualLicenceCount, 0) >= coalesce(Y.requiredLicenceCount, 0) and PCCC.status='approved' then 'C'
                            else
                                case
                                    when PendingCerts.startedLicenceCount > 0 then 'S'
                                    when PendingCerts.submittedLicenceCount > 0 then 'SU'
                                    when PendingCerts.returnedLicenceCount > 0 then 'R'
                                    else 'X'
                                end
                        end licenceCompliant,
                         X.actualLicenceCount, Y.requiredLicenceCount,
                        case
                            when PCCC.status='started' then 'S'
                            when PCCC.status = 'submitted' then 'SU'
                            when PCCC.status = 'returned' then 'R'
                            when CCS.pendingCompanyComplianceCertsId is null or coalesce(Y.requiredInsuranceCount, 0) = 0 and PCCC.status='approved' then 'N/A'
          
                            when X.minInsuranceExpiryDays<30 and X.actualInsuranceCount=Y.requiredInsuranceCount and PCCC.status='approved' then '1'
                            when X.minInsuranceExpiryDays<60 and X.actualInsuranceCount=Y.requiredInsuranceCount and PCCC.status='approved' then '2'
                            when X.minInsuranceExpiryDays<90 and X.actualInsuranceCount=Y.requiredInsuranceCount and PCCC.status='approved' then '3'
                            when CCS.pendingCompanyComplianceCertsId >0 and coalesce(X.actualInsuranceCount, 0) >= coalesce(Y.requiredInsuranceCount, 0) and PCCC.status='approved' then 'C'
                            else
                                case
                                    when PendingCerts.startedInsuranceCount > 0 then 'S'
                                    when PendingCerts.submittedInsuranceCount > 0 then 'SU'
                                    when PendingCerts.returnedInsuranceCount > 0 then 'R'
                                    else 'X'
                                end
                        end insuranceCompliant,
                   
                        PCCC.status insuranceCompliantStatus,X.actualInsuranceCount, Y.requiredInsuranceCount,
                        case
                            when CCS.pendingCompanyComplianceSMSReviewId is null and Z.companyId is null then ''
                          
                            when Z.SmsExpiryDays<30 and Z.coCertId is not null and Z.status='approved'  then '1'
                            when Z.SmsExpiryDays<60 and Z.coCertId is not null and Z.status='approved'  then '2'
                            when Z.SmsExpiryDays<90 and Z.coCertId is not null and Z.status='approved'  then '3'
                            when Z.status='started' then 'S'
                            when Z.status='submitted' then 'SU'
                            when Z.status='return' then 'R'
                            when Z.coCertId is not null and Z.status='approved'
                                and now() between Z.certStartDate and Z.certEndDate
                                and Z.certActive='y'
                                then 'C'
                            else 'X'
